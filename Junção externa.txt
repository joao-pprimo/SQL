	*Junção externa*

- Sintaxe básica 

junção à esquerda:
		  SELECT C.CODIGOCURSO, C.NOME,
			N.CODIGONIVEL, N.DESCRICAO
		   FROM CURSO C LEFT JOIN NIVEL N ON (N.CODIGONIVEL=C.CODIGONIVEL) ; 

junção à direita:
		  SELECT C.CODIGOCURSO, C.NOME,
 			N.CODIGONIVEL, N.DESCRICAO
		  FROM CURSO C RIGHT JOIN NIVEL N ON
		  (N.CODIGONIVEL=C.CODIGONIVEL);
